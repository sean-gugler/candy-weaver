0 REM GAME BY SEAN GUGLER
10 DIM C(15)
20 GR
30 W=6:H=8:X=14:Y=4
40 CN%=16:GOSUB 100
50 GOSUB 200
60 END
100 REM GENERATE COLOR PALETTE
110 FOR CI=0 TO CN%-1:C(CI)=RND(1)*15+1:NEXT:RETURN
200 REM DRAW PATCH
210 CI%=0
220 FOR BI=0 TO W+H-1
230 COLOR=C(CI%)
240 GOSUB 300
250 CI%=CI%+1:IF CI%=CN% THEN CI%=0
260 NEXT BI
270 RETURN
300 REM DRAW A SINGLE BAR
310 IF BI<W THEN VLIN Y,Y+H*2 AT 1+X+BI*2:RETURN
320 HLIN X,X+W*2 AT 1+Y+(BI-W)*2:RETURN
